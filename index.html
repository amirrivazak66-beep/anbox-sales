function downloadExcelTemplate() { const headers = [ 'نوع لید', 'مبدأ لید', 'نام شرکت', 'صنعت', 'دلیل انتخاب شرکت', 'شماره تماس', 'نام فرد تماس', 'سمت فرد تماس', 'مرحله فروش', 'تناسب با آن باکس', 'مانع اصلی', 'اقدام بعدی', 'کارشناس فروش', 'وضعیت لید', 'یادداشت‌ها' ]; let csv = headers.join('\t') + '\n'; const blob = new Blob([csv], { type: 'text/csv;charset=utf-8;' }); const link = document.createElement('a'); const url = URL.createObjectURL(blob); link.setAttribute('href', url); link.setAttribute('download', 'نمونه_لیدها.csv'); link.style.visibility = 'hidden'; document.body.appendChild(link); link.click(); document.body.removeChild(link); showNotification('✅ فایل نمونه دانلود شد'); }
